⏺ Sektion 6: Säkerhet, Begränsningar & Guardrails                                                                                                                                                                                      
                                                                                                                                                                                                                                       
  SQL Injection-skydd                                            
                                                                                                                                                                                                                                       
  Claude genererar SQL som körs mot kundens databas. Det här kräver strikta guardrails:                                                                                                                                                
                                                                 
  // lib/claude/tools.ts — execute_sql tool                                                                                                                                                                                            
                                                
  async function executeSql(tenantClient: SupabaseClient, query: string) {                                                                                                                                                             
    // 1. Bara SELECT tillåts                  
    const normalized = query.trim().toUpperCase()                                                                                                                                                                                      
    if (!normalized.startsWith('SELECT')) {     
      throw new Error('Only SELECT queries allowed via execute_sql')                                                                                                                                                                   
    }                                           
                                                                                                                                                                                                                                       
    // 2. Blocka farliga mönster                
    const blocked = ['DROP', 'DELETE', 'TRUNCATE', 'ALTER', 'CREATE', 'INSERT', 'UPDATE', 'GRANT', 'REVOKE']                                                                                                                           
    for (const word of blocked) {                                                                                                                                                                                                      
      if (normalized.includes(word)) {                                                                                                                                                                                                 
        throw new Error(`Blocked keyword: ${word}`)                                                                                                                                                                                    
      }                                                                                                                                                                                                                                
    }                                                                                                                                                                                                                                  
                                                                                                                                                                                                                                       
    // 3. Kör via read-only connection (Supabase service role med RLS)                                                                                                                                                                 
    // 4. Timeout: 10 sekunder max                                                                                                                                                                                                     
    // 5. Row limit: max 1000 rader                                                                                                                                                                                                    
    const { data, error } = await tenantClient.rpc('execute_readonly_query', {                                                                                                                                                         
      sql: query,                                                                                                                                                                                                                      
      row_limit: 1000                                                                                                                                                                                                                  
    })                                                                                                                                                                                                                                 
                                                                                                                                                                                                                                       
    return data                                                                                                                                                                                                                        
  }
